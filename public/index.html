    <script>

        document.addEventListener('DOMContentLoaded', function() {

            // Store quiz answers

            const quizData = {

                answers: {},

                userInfo: {}

            };


            // Variables to track which question is currently shown

            let currentQuestion = 1;

            let totalQuestions = 15; // Including the loading and lead form screens


            // Get all question slides

            const questionSlides = document.querySelectorAll('.question-slide');

            const progressBar = document.getElementById('progress-bar');


            // Update progress bar

            function updateProgressBar() {

                const progress = (currentQuestion / totalQuestions) * 100;

                progressBar.style.width = `${progress}%`;

            }


            // Show a specific question slide

            function showQuestion(questionNumber) {

                // Hide all questions

                questionSlides.forEach(slide => {

                    slide.classList.remove('active');

                });


                // Show the current question

                const currentSlide = document.querySelector(`#question-${questionNumber}`);

                if (currentSlide) {

                    currentSlide.classList.add('active');

                } else {

                    // If we're at a special screen

                    if (questionNumber === 13) {

                        document.getElementById('loading-screen').classList.add('active');

                    } else if (questionNumber === 14) {

                        document.getElementById('lead-form').classList.add('active');

                    } else if (questionNumber === 15) {

                        document.getElementById('otp-screen').classList.add('active');

                    }

                }


                currentQuestion = questionNumber;

                updateProgressBar();

            }


            // Handle navigation between questions

            function navigateToNextQuestion() {

                // Check for conditional logic

                if (currentQuestion === 2) {

                    const hasInsurance = quizData.answers[2];

                    

                    // If "No" to insurance, skip questions 3, 4, 5

                    if (hasInsurance === "No") {

                        showQuestion(6);

                    } else {

                        showQuestion(currentQuestion + 1);

                    }

                } else {

                    showQuestion(currentQuestion + 1);

                }

            }


            // Go back to previous question

            function navigateToPreviousQuestion() {

                // Check for conditional logic when going back

                if (currentQuestion === 6 && quizData.answers[2] === "No") {

                    showQuestion(2);

                } else {

                    showQuestion(currentQuestion - 1);

                }

            }


            // Check for disqualification

            function checkDisqualification() {

                // Age is 65+

                if (quizData.answers[6] === "65+") {

                    window.location.href = "https://yourlandingpage.com/sorry";

                    return true;

                }

                

                // Employment status is Unemployed

                if (quizData.answers[8] === "Unemployed") {

                    window.location.href = "https://yourlandingpage.com/sorry";

                    return true;

                }

                

                return false;

            }


            // Add click event for option buttons

            document.querySelectorAll('.option').forEach(option => {

                option.addEventListener('click', function() {

                    const questionId = this.closest('.question-slide').getAttribute('data-question');

                    const answer = this.getAttribute('data-value');

                    

                    // Store the answer

                    quizData.answers[questionId] = answer;

                    

                    // Check for disqualification

                    if (checkDisqualification()) {

                        return; // Stop if disqualified

                    }

                    

                    // Navigate to next question

                    navigateToNextQuestion();

                });

            });


            // Add click event for back buttons

            document.querySelectorAll('.btn-back').forEach(button => {

                button.addEventListener('click', navigateToPreviousQuestion);

            });


            // Handle multi-select questions

            document.querySelectorAll('#next-multi-select').forEach(button => {

                button.addEventListener('click', function() {

                    const questionId = this.closest('.question-slide').getAttribute('data-question');

                    const checkboxes = this.closest('.question-slide').querySelectorAll('input[type="checkbox"]:checked');

                    

                    if (checkboxes.length === 0) {

                        // No option selected, show error

                        alert('Please select at least one option');

                        return;

                    }

                    

                    // Store all selected values

                    const selectedValues = Array.from(checkboxes).map(checkbox => checkbox.value);

                    quizData.answers[questionId] = selectedValues;

                    

                    // Navigate to next question

                    navigateToNextQuestion();

                });

            });


            // Handle checkbox styling

            document.querySelectorAll('.checkbox-option').forEach(option => {

                option.addEventListener('click', function(e) {

                    if (e.target.tagName !== 'INPUT') {

                        const checkbox = this.querySelector('input[type="checkbox"]');

                        checkbox.checked = !checkbox.checked;

                    }

                    

                    // Toggle selected class

                    if (this.querySelector('input[type="checkbox"]').checked) {

                        this.classList.add('selected');

                    } else {

                        this.classList.remove('selected');

                    }

                });

            });


            // Handle occupation (free text) submission

            document.getElementById('next-occupation').addEventListener('click', function() {

                const occupation = document.getElementById('occupation').value.trim();

                const errorElement = document.getElementById('occupation-error');

                

                if (occupation === '') {

                    errorElement.textContent = 'Please enter your occupation';

                    return;

                }

                

                errorElement.textContent = '';

                quizData.answers[12] = occupation;

                

                // Show loading screen for 3 seconds before showing lead form

                showQuestion(13);

                setTimeout(() => {

                    showQuestion(14);

                }, 3000);

            });
